pipeline {

  agent {

    docker {

      image 'windsekirun/jenkins-android-docker:1.1.1'

    }

  }

  options {

    // Stop the build early in case of compile or test failures

    skipStagesAfterUnstable()

  }

  stages {

    stage ('Prepare'){

      steps {

     

      sh 'cd gradle-site-plugin'
        sh 'chmod +x ./gradle-site-plugin/gradlew'

      }

    }

    stage('Compile') {

      steps {

        // Compile the app and its dependencies

        sh './gradle-site-plugin/gradlew compileDebugSources'

      }

    }

    stage('Build APK') {

      steps {

      

        // Finish building and packaging the APK

        sh './gradle-site-plugin/gradlew assembleDebug'

      }

    }

  }

}
